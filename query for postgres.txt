CREATE TABLE news (
    id  SERIAL PRIMARY KEY,
    title varchar(255) NOT NULL,
    description text NOT NULL,
    full_news text NOT NULL,
    time date NOT NULL
)


CREATE TABLE files(
    id  SERIAL,
    file_name text NOT NULL PRIMARY KEY,
	file_type text 
)

CREATE TABLE panorama(
    id  SERIAL PRIMARY KEY,
    img_name text NOT NULL,
	constraint fk_tests_students
     foreign key (img_name) 
     REFERENCES files (file_name),
    active text
)


CREATE TABLE achizitii_recente(
    id  SERIAL PRIMARY KEY,
	time Date NOT NULL,
    imagesPath text NOT NULL,
    list text NOT NULL
)



CREATE TABLE program (
    id  SERIAL PRIMARY KEY,
    days varchar(255) NOT NULL,
    time varchar(255) NOT NULL
)


CREATE TABLE servicii (
    id  SERIAL PRIMARY KEY,
    link_title text NOT NULL,
    content text 
)

CREATE TABLE resurse (
    id  SERIAL PRIMARY KEY,
    link_title text NOT NULL,
    content text 
)

CREATE TABLE produse (
    id  SERIAL PRIMARY KEY,
    link_title text NOT NULL,
    content text 
)



CREATE TABLE proiecte(
    id  SERIAL PRIMARY KEY,
    img_name text NOT NULL,
	constraint fk_tests
    foreign key (img_name) 
    REFERENCES files (file_name),
    description text,
	link text
)

CREATE TABLE contacte (
    id  SERIAL PRIMARY KEY,
    contact_name text NOT NULL,
    contact_description text NOT NULL
)


CREATE TABLE linkuri(
    id  SERIAL PRIMARY KEY,
    img_name text NOT NULL,
	constraint fk_tests
    foreign key (img_name) 
    REFERENCES files (file_name),
    description text,
	link text
)


CREATE TABLE despre_noi (
    id  SERIAL PRIMARY KEY,
    title text NOT NULL,
    content text 
)

CREATE TABLE tutoriale (
    id  SERIAL PRIMARY KEY,
    title text NOT NULL,
    content text 
)

CREATE TABLE pentru_utilizatori (
    id  SERIAL PRIMARY KEY,
    title text NOT NULL,
    content text 
)

CREATE TABLE header (
    id  SERIAL PRIMARY KEY,
    logo text NOT NULL,
	constraint fk_tests_logo
    foreign key (logo)
	REFERENCES files (file_name),
    slogan text 
)
